# Define a function to calculate sum of squares of even numbers up to n
function sum_of_squares_of_evens(n::Int)
    # Define the range of numbers
    numbers = 1:n
    
    # Filter for even numbers
    even_numbers = filter(x -> x % 2 == 0, numbers)
    
    # Square each even number
    squared_evens = [x^2 for x in even_numbers]
    
    # Calculate the sum of the squared values
    result = sum(squared_evens)
    
    return result
end
